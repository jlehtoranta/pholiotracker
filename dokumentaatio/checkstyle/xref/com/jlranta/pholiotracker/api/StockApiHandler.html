3   import java.util.ArrayList;
4   import java.util.TreeMap;
5   
6   public class StockApiHandler {
7       private ArrayList<StockApi> stockApis = new ArrayList<>();
8       
9       public StockApiHandler() {
10          this.initializeApis();
11      }
12      
13      public TreeMap<StockApi, ArrayList<String>> searchAll(String s) {
14          TreeMap<StockApi, ArrayList<String>> foundFromApis = new TreeMap<>();
15          
16          for (StockApi api : this.stockApis) {
17              ArrayList<String> found = api.search(s);
18              if (!found.isEmpty()) {
19                  foundFromApis.put(api, found);
20              }
21          }
22          
23          return foundFromApis;
24      }
25      
26      public ArrayList<StockApi> getApis() {
27          return this.stockApis;
28      }
29      
30      private void initializeApis() {
31          StockApi vantage = new AlphaVantageApi();
32          this.stockApis.add(vantage);
33      }
34  }
