3   import java.util.ArrayList;
4   import java.util.LinkedHashMap;
5   
6   /**
7    * Initializes and handles all the APIs that implement the StockApi interface
8    * @author Jarkko Lehtoranta <devel@jlranta.com>
9    */
10  public class StockApiHandler {
11      private ArrayList<StockApi> stockApis = new ArrayList<>();
12      
13      public StockApiHandler() {
14          this.initializeApis();
15      }
16      
17      /**
18       * 
19       * @param s A company name or a stock symbol
20       * @return  A LinkedHashMap of StockApis as keys and a list of found
21       *          stock symbols as strings
22       */
23      public LinkedHashMap<StockApi, ArrayList<String>> searchAll(String s) {
24          LinkedHashMap<StockApi, ArrayList<String>> foundFromApis = new LinkedHashMap<>();
25          
26          for (StockApi api : this.stockApis) {
27              ArrayList<String> found = api.search(s);
28              if (!found.isEmpty()) {
29                  foundFromApis.put(api, found);
30              }
31          }
32          
33          return foundFromApis;
34      }
35      
36      public ArrayList<StockApi> getApis() {
37          return this.stockApis;
38      }
39      
40      private void initializeApis() {
41          StockApi vantage = new AlphaVantageApi();
42          this.stockApis.add(vantage);
43      }
44  }
