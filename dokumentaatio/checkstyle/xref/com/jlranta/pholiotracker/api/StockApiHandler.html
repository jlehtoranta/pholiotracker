3   import java.util.ArrayList;
4   
5   /**
6    * Initializes and handles all the APIs that implement the StockApi interface.
7    * @author Jarkko Lehtoranta
8    */
9   public class StockApiHandler {
10      private ArrayList<StockApi> stockApis = new ArrayList<>();
11      private StockSearchApi stockSearch;
12      
13      /**
14       * Creates a new StockApiHandler instance.
15       */
16      public StockApiHandler() {
17          this.initializeApis();
18      }
19      
20      /**
21       * Search for a stock.
22       * @param s A part of a stock name or a stock symbol
23       * @return  A list of StockSearchResults
24       */
25      public ArrayList<StockSearchResult> searchStock(String s) {
26          return this.stockSearch.search(s);
27      }
28      
29      /**
30       * Get the most recent price quote for a stock.
31       * @param s    A stock symbol
32       * @param exch A stock exchange, where the symbol is listed
33       * @return     The most recent price as a StockSearchResult
34       */
35      public StockSearchResult getCurrentPrice(String s, String exch) {
36          StockSearchResult res = null;
37          
38          for (StockApi api : this.stockApis) {
39              if (api.toString().equals("Alpha Vantage")) {
40                  res = api.getQuote(s, exch);
41              }
42          }
43          
44          return res;
45      }
46      
47      /**
48       * Search for a stock API, which provides data for a stock symbol.
49       * @param s A company name or a stock symbol
50       * @return  A LinkedHashMap of StockApis as keys and a list of found
51       *          stock symbols as strings
52       */
53      public StockApi searchApi(String s) {
54  
55          for (StockApi api : this.stockApis) {
56              ArrayList<String> found = api.search(s);
57              if (!found.isEmpty()) {
58                  return api;
59              }
60          }
61          
62          return null;
63      }
64      
65      public ArrayList<StockApi> getApis() {
66          return this.stockApis;
67      }
68      
69      private void initializeApis() {
70          this.stockSearch = new StockSearchApi();
71          
72          StockApi vantage = new AlphaVantageApi();
73          this.stockApis.add(vantage);
74      }
75  }
