3   import java.io.IOException;
4   import org.apache.http.HttpEntity;
5   import org.apache.http.HttpResponse;
6   import org.apache.http.ParseException;
7   import org.apache.http.client.methods.HttpGet;
8   import org.apache.http.impl.client.CloseableHttpClient;
9   import org.apache.http.impl.client.HttpClients;
10  import org.apache.http.util.EntityUtils;
11  
12  public class RestApiQuery {
13      private String apiName = "";
14      private CloseableHttpClient cli;
15      
16      public RestApiQuery(String apiName) {
17          this.apiName = apiName;
18          this.cli = HttpClients.createDefault();
19      }
20      
21      public String apiRequest(String url) {
22          String body = "";
23          
24          try {
25              HttpGet get = new HttpGet(url);
26              HttpResponse r = this.cli.execute(get);
27              
28              if (r.getStatusLine().getStatusCode() != 200) {
29                  System.err.println(this.apiName + " query failed on HTTP error (" + r.getStatusLine().getStatusCode() + ")");
30              } else {
31                  HttpEntity entity = r.getEntity();
32                  if (entity != null) {
33                      body = EntityUtils.toString(entity);
34                  } else {
35                      System.err.println(this.apiName + " empty HTTP response");
36                  }
37              }
38          } catch (IOException | ParseException e) {
39              System.err.println(this.apiName + " HTTP request failed");
40          }
41          
42          return body;
43      } 
44  }
