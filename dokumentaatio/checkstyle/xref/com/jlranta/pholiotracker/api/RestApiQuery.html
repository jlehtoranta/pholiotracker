3   import java.io.IOException;
4   import org.apache.http.HttpEntity;
5   import org.apache.http.HttpResponse;
6   import org.apache.http.ParseException;
7   import org.apache.http.client.methods.HttpGet;
8   import org.apache.http.impl.client.CloseableHttpClient;
9   import org.apache.http.impl.client.HttpClients;
10  import org.apache.http.util.EntityUtils;
11  
12  /**
13   * RestApiQuery class is a helper class for placing HTTP queries to the stock
14   * APIs. It uses org.apache.http.* for querying.
15   * @author Jarkko Lehtoranta <devel@jlranta.com>
16   */
17  public class RestApiQuery {
18      private String apiName = "";
19      private CloseableHttpClient cli;
20      
21      /**
22       * @param apiName The display name of the API, where the queries come from
23       */
24      public RestApiQuery(String apiName) {
25          this.apiName = apiName;
26          this.cli = HttpClients.createDefault();
27      }
28      
29      /**
30       * A simple methods that handles the generic HTTP GET requests
31       * @param url The URL for the HTTP GET request
32       * @return The body of the HTTP response as a string
33       */
34      public String apiRequest(String url) {
35          String body = "";
36          
37          try {
38              HttpGet get = new HttpGet(url);
39              HttpResponse r = this.cli.execute(get);
40              
41              if (r.getStatusLine().getStatusCode() != 200) {
42                  System.err.println(this.apiName + " query failed on HTTP error (" + r.getStatusLine().getStatusCode() + ")");
43              } else {
44                  HttpEntity entity = r.getEntity();
45                  if (entity != null) {
46                      body = EntityUtils.toString(entity);
47                  } else {
48                      System.err.println(this.apiName + " empty HTTP response");
49                  }
50              }
51          } catch (IOException | ParseException e) {
52              System.err.println(this.apiName + " HTTP request failed");
53          }
54          
55          return body;
56      } 
57  }
